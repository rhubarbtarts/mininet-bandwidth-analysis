# Mininet Bandwidth Measurement - Commands

# Update system

sudo apt update

# Install required tools

sudo apt install mininet iperf -y

# Clean previous Mininet configuration

sudo mn -c

# Start Mininet (default topology)

sudo mn

# Test connectivity

pingall

# Bandwidth test (Topology 1: 2 hosts)

h1 iperf -s &
h2 iperf -c h1

# Exit Mininet

exit

# Start Mininet with 3 hosts

sudo mn --topo single,3

# (Optional) Test connectivity again

pingall

# Bandwidth test (Topology 2: 3 hosts)

h1 iperf -s &
h3 iperf -c h1

# Exit Mininet

exit
